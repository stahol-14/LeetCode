class Solution {
    public int maxSubArray(int[] nums) {
        int n = nums.length;
        int[] dp = new int[n];
        dp[0] = nums[0];
        int res = dp[0];

        for(int i = 1; i < n; i++){
            dp[i] = Math.max(dp[i - 1] + nums[i], nums[i]); // è¦ä¹ˆç»§ç»­ç´¯åŠ ï¼Œè¦ä¹ˆé‡æ–°å¼€å§‹
            res = Math.max(dp[i], res);
        }

        return res;
    }
}
