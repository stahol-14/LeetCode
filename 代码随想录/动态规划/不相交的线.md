    public int maxUncrossedLines(int[] nums1, int[] nums2) {
        int len1 = nums1.length;
        int len2 = nums2.length;

            for(int j = 1; j <= len2; j++){
                int b = nums2[j - 1];
                if(a == b){
                    dp[i][j] = dp[i - 1][j - 1] + 1;

            }
        }

        return dp[len1][len2];
    }
