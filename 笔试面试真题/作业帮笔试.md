class Solution {
    public int maxArea(int[] height) {
        int left = 0, right = height.length - 1;
        int res = 0;

        while(left < right){
            int width = Math.min(height[left], height[right]);
            int len = right - left;
            int area = width * len;

            res = Math.max(res, area);

            if(height[left] < height[right]){
                left++;
            }else{
                right--;
            }
        }

        return res;
    }
}
